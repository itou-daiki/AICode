📋 リスト (List)

順序のあるデータの集合。要素の追加・削除・変更が可能です。

# リストの作成 numbers = [1, 2, 3, 4, 5] mixed = ["文字", 123, True, [1, 2]] # 要素のアクセス print(numbers[0]) # 1 (最初の要素) print(numbers[-1]) # 5 (最後の要素) print(numbers[1:4]) # [2, 3, 4] (スライス) # 要素の追加・削除 numbers.append(6) # 末尾に追加 numbers.insert(0, 0) # 先頭に追加 numbers.remove(3) # 値3を削除 popped = numbers.pop() # 末尾を削除して取得 # リスト内包表記 squares = [x**2 for x in range(5)] # [0, 1, 4, 9, 16]

📚 辞書 (Dictionary)

キーと値のペアでデータを管理します。

# 辞書の作成 student = { "name": "田中", "age": 18, "grade": "高校3年", "subjects": ["数学", "英語", "国語"] } # 値のアクセス・変更 print(student["name"]) # 田中 print(student.get("age")) # 18 student["age"] = 19 # 値を変更 student["hobby"] = "読書" # 新しいキーを追加 # 辞書の操作 print(student.keys()) # キー一覧 print(student.values()) # 値一覧 print(student.items()) # キー・値のペア # 辞書内包表記 word_lengths = {word: len(word) for word in ["Python", "Java", "C++"]}

🔢 セット (Set)

重複のない要素の集合。集合演算が可能。

# セットの作成 fruits = {"apple", "banana", "orange", "apple"} # 重複は削除 print(fruits) # {'banana', 'orange', 'apple'} # 集合演算 set1 = {1, 2, 3, 4} set2 = {3, 4, 5, 6} print(set1 & set2) # {3, 4} (積集合) print(set1 | set2) # {1, 2, 3, 4, 5, 6} (和集合) print(set1 - set2) # {1, 2} (差集合)

🔧 関数とクラス

⚙️ 関数の定義

# 基本的な関数 def greet(name): return f"こんにちは、{name}さん！" # デフォルト引数 def introduce(name, age=20): return f"私は{name}、{age}歳です" # 複数の戻り値 def calculate(a, b): return a + b, a - b, a * b, a / b # 可変長引数 def sum_all(*numbers): return sum(numbers) # キーワード引数 def create_profile(**kwargs): return kwargs # 使用例 print(greet("太郎")) print(introduce("花子")) add, sub, mul, div = calculate(10, 3) print(sum_all(1, 2, 3, 4, 5)) profile = create_profile(name="佐藤", age=30, city="東京")

🏗️ クラスとオブジェクト

class Student: # クラス変数 school_name = "Python高校" def __init__(self, name, age): # インスタンス変数 self.name = name self.age = age self.grades = [] def add_grade(self, subject, score): self.grades.append({"subject": subject, "score": score}) def get_average(self): if not self.grades: return 0 total = sum(grade["score"] for grade in self.grades) return total / len(self.grades) def __str__(self): return f"Student: {self.name} ({self.age}歳)" # クラスの使用 student = Student("田中太郎", 17) student.add_grade("数学", 85) student.add_grade("英語", 92) print(student) print(f"平均点: {student.get_average()}")

🔒 継承とカプセル化

class Person: def __init__(self, name, age): self.name = name self._age = age # プロテクト属性 self.__id = id(self) # プライベート属性 def speak(self): return f"{self.name}が話しています" class Teacher(Person): # Personクラスを継承 def __init__(self, name, age, subject): super().__init__(name, age) # 親クラスの初期化 self.subject = subject def teach(self): return f"{self.name}先生が{self.subject}を教えています" def speak(self): # メソッドのオーバーライド return f"{self.name}先生が授業中です" # 使用例 teacher = Teacher("山田花子", 35, "数学") print(teacher.teach()) print(teacher.speak())

📊 データ処理の例

# 学生の成績データを処理 students_data = [ {"name": "田中", "math": 85, "english": 78, "science": 92}, {"name": "佐藤", "math": 92, "english": 85, "science": 88}, {"name": "鈴木", "math": 78, "english": 92, "science": 85}, ] # 各学生の平均点を計算 def calculate_averages(students): for student in students: subjects = ["math", "english", "science"] total = sum(student[subject] for subject in subjects) student["average"] = round(total / len(subjects), 1) return students # 上位学生を抽出 def get_top_students(students, threshold=85): return [s for s in students if s["average"] >= threshold] # 実行 students_with_avg = calculate_averages(students_data) top_students = get_top_students(students_with_avg) for student in top_students: print(f"{student['name']}: 平均{student['average']}点")

🎯 簡単なゲーム

import random class NumberGuessGame: def __init__(self, max_number=100): self.target = random.randint(1, max_number) self.attempts = 0 self.max_attempts = 7 def play(self): print(f"1から100の数を当ててください！（{self.max_attempts}回まで）") while self.attempts < self.max_attempts: try: guess = int(input("予想: ")) self.attempts += 1 if guess == self.target: print(f"正解！{self.attempts}回で当てました！") return True elif guess < self.target: print("もっと大きい数です") else: print("もっと小さい数です") remaining = self.max_attempts - self.attempts print(f"残り{remaining}回") except ValueError: print("数値を入力してください") print(f"ゲームオーバー！答えは{self.target}でした") return False # ゲーム実行 # game = NumberGuessGame() # game.play()
