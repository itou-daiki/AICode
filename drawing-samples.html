描画モード - サンプルコード集

p5.jsライクなPython描画機能のサンプルコードです。

基本図形

# 円を描画
p5.circle(200, 200, 100)

# 四角形を描画
p5.rect(150, 150, 100, 50)

# 線を描画
p5.line(50, 50, 350, 350)

色の設定

# 赤色で塗りつぶし
p5.fill(255, 0, 0)
p5.circle(150, 150, 80)

# 緑色の輪郭
p5.stroke(0, 255, 0)
p5.stroke_weight(3)
p5.rect(250, 150, 80, 80)

複雑な図形

# 三角形
p5.triangle(200, 100, 150, 200, 250, 200)

# 楕円
p5.ellipse(200, 300, 120, 80)

# 弧
import math
p5.arc(200, 200, 100, 100, 0, math.pi)

アニメーション風の描画

# 背景をクリア
p5.background(240, 240, 240)

# 複数の円を描画
for i in range(5):
    x = 80 + i * 60
    y = 200
    p5.fill(255 - i * 50, i * 50, 100)
    p5.circle(x, y, 40)

座標変換

# 座標系を保存
p5.push()

# 中心に移動して回転
p5.translate(200, 200)
p5.rotate(45)  # 45度回転（度数法）

# 四角形を描画
p5.rect(-25, -25, 50, 50)

# 座標系を復元
p5.pop()

テキスト描画

# テキストサイズを設定
p5.text_size(24)

# テキストを描画
p5.fill(0, 0, 255)
p5.text("Hello, Python!", 150, 200)

パターン描画

# グリッドパターン
for x in range(0, 400, 40):
    for y in range(0, 400, 40):
        if (x + y) % 80 == 0:
            p5.fill(200, 100, 100)
        else:
            p5.fill(100, 100, 200)
        p5.rect(x, y, 40, 40)

数学的な図形

# スパイラル
import math

p5.stroke(100, 50, 200)
p5.stroke_weight(2)
p5.no_fill()

for i in range(100):
    angle = i * 0.2
    radius = i * 2
    x = 200 + radius * math.cos(angle)
    y = 200 + radius * math.sin(angle)
    p5.point(x, y)
